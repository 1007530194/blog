import folium
from folium.features import DivIcon

m = folium.Map(
    location=[31, 110],
    zoom_start=5,
    #tiles='Mapbox Bright'
)

d5=[]
for i in range(4,9):
    for j in range(21,28):
        d5.append((i,j,str(i)+'_'+str(j)))

#d5 = [(6,23,'1_2')]

for line in d5:
    
    p1 = [float(line[0])*5, float(line[1])*5]
    folium.Marker(p1, icon=DivIcon(
            icon_size=(150,36),
            icon_anchor=(7,20),
            html='<div style="font-size: 15pt; color : black">'+line[2]+'</div>',
            )).add_to(m)
    m.add_child(folium.CircleMarker(p1, radius=15))
m
